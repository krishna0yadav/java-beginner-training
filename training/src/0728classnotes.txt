1.Write validation functions and modify the onSubmit event Handler in the form code to validate the following form fields:
1. Member number
• Must be a number
2. Password
• Must be longer than 4 characters
If an error occurs, use an alert box to display an error message for the first field in error and place the cursor in the first field in error.


<html>
<head>
	<script>
		function show()
		{
			var a = document.getElementById('num1').value;
			var b = document.getElementById('pass').value;
			
			var pat1= /^\d+$/;
			var pat2= /\w{4}/;
			if(a.match(pat1))
			{
				if(b.match(pat2))
				{
					document.getElementById('sp1').innerHTML="Valid";
				}
				else
				{
					document.getElementById('sp1').innerHTML="Invalid";
					document.getElementById('pass').select();
				}
			}
			else
			{
				document.getElementById('sp1').innerHTML="Invalid";
				document.getElementById('num1').select();
			}
		}
	
	</script>
</head>
<body>
<form>
	<input type="text" id="num1" />
	<input type="password" id="pass" />
	<input type="button" value="show" onclick="show()"/>

</form>

</body>
</html>

3. Write a validation function to check a password between 6 to 20 characters which contain at least one numeric digit, one uppercase and one lowercase 
[Hint use: ?=]

how are you
/are (?= you)/
<html>
<head>
	<script>
		function show()
		{
			var a = document.getElementById('num1').value;
			var pat1 = /(?=.*\d)(?=.*[a-z])(?=.*[A-Z]).{6,20}/;
			
			if(a.match(pat1))
			{
					document.getElementById('sp1').innerHTML="Valid";
			}
			else
			{
				document.getElementById('sp1').innerHTML="Invalid";
				
			}
		}
	
	</script>
</head>
<body>
<form>
	<input type="text" id="num1" />

</form>
